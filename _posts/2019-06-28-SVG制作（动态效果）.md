  height: 50%;

</style>

             attributeName="d" 
             dur="3000ms" 
             repeatCount="indefinite"
             values="M 0,0

                     50,100 50,100 0,100
                     0,50 0,50 0,0

                     M 25,50 
                     C 37.5,25 37.5,25 50,0 
                     75,50 75,50 100,100
                     50,100 50,100 0,100

  </path>

</div>
